120  static void YOLOPreProcessor(const DatasetY& annotations,
121  arma::Mat<eT>& output,
122  const size_t version = 1,
123  const size_t imageWidth = 224,
124  const size_t imageHeight = 224,
125  const size_t gridWidth = 7,
126  const size_t gridHeight = 7,
127  const size_t numBoxes = 2,
128  const size_t numClasses = 20,
129  const bool normalize = true)
130  {
131  // See if we can change this to v4 / v5.
132  mlpack::Log::Assert(version >= 1 && version <= 3, "Supported YOLO versions \
133  are version 1 to version 3.");
134 
135  mlpack::Log::Assert(typeid(annotations) == typeid(arma::field<arma::vec>),
136  "Use Field type to represent annotations.");
137 
138  size_t batchSize = annotations.n_cols;
139  size_t numPredictions = 5 * numBoxes + numClasses;
140  if (version > 1)
141  {
142  // Each bounding boxes has a corresponding class.
143  numPredictions = numBoxes * (5 + numClasses);
144  }
145 
146  double cellSizeHeight = (double) 1.0 / gridHeight;
147  double cellSizeWidth = (double) 1.0 / gridWidth;
148 
149  // Set size of output and use cubes convenience.
150  output.set_size(gridWidth * gridHeight * numPredictions, batchSize);
151  output.zeros();
152 
153  // Use offset to create a cube for a particular column / batch.
154  size_t offset = 0;
155  for (size_t boxIdx = 0; boxIdx < batchSize; boxIdx++)
156  {
157  arma::cube outputTemp(const_cast<arma::Mat<eT> &>(output).memptr() +
158  offset, gridHeight, gridWidth, numPredictions, false, false);
159  offset += gridWidth * gridHeight * numPredictions;
160 
161  // Get the bounding box and labels corresponding to current image.
162  arma::mat labels(1, annotations(0, boxIdx).n_elem / 5);
163  arma::mat boundingBoxes(4, annotations(0, boxIdx).n_elem / 5);
164  for (size_t i = 0; i < boundingBoxes.n_cols; i++)
165  {
166  labels.col(i)(0) = annotations(0, boxIdx)(i * 5);
167  boundingBoxes.col(i) = annotations(0, boxIdx)(arma::span(i * 5 + 1,
168  (i + 1) * 5 - 1));
169  }
170 
171  // For YOLOv2 or higher, each bounding box can represent a class
172  // so we don't repeat labels as done for YOLOv1. We will use map
173  // to store last inserted bounding box.
174  std::map<std::pair<size_t, size_t>, size_t> boundingBoxOffset;
175 
176  // Normalize the coordinates.
177  boundingBoxes.row(0) /= imageWidth;
178  boundingBoxes.row(2) /= imageWidth;
179  boundingBoxes.row(1) /= imageHeight;
180  boundingBoxes.row(3) /= imageHeight;
181 
182  // Get width and height as well as centres for the bounding box.
183  arma::mat widthAndHeight(2, boundingBoxes.n_cols);
184  widthAndHeight.row(0) = (boundingBoxes.row(2) - boundingBoxes.row(0));
185  widthAndHeight.row(1) = (boundingBoxes.row(3) - boundingBoxes.row(1));
186 
187  arma::mat centres(2, boundingBoxes.n_cols);
188  centres.row(0) = (boundingBoxes.row(2) + boundingBoxes.row(0)) / 2.0;
189  centres.row(1) = (boundingBoxes.row(3) + boundingBoxes.row(1)) / 2.0;
190 
191  // Assign bounding boxes to the grid.
192  for (size_t i = 0; i < boundingBoxes.n_cols; i++)
193  {
194  // Index for representing bounding box on grid.
195  arma::vec gridCoordinates = centres.col(i);
196  arma::vec centreCoordinates = centres.col(i);
197 
198  if (normalize)
199  {
200  gridCoordinates(0) = std::ceil(gridCoordinates(0) /
201  cellSizeWidth) - 1;
202  gridCoordinates(1) = std::ceil(gridCoordinates(1) /
203  cellSizeHeight) - 1;
204  }
205  else
206  {
207  gridCoordinates(0) = std::ceil((gridCoordinates(0) /
208  imageWidth) / cellSizeWidth) - 1;
209  gridCoordinates(1) = std::ceil((gridCoordinates(1) /
210  imageHeight) / cellSizeHeight) - 1;
211  }
212 
213  size_t gridX = gridCoordinates(0);
214  size_t gridY = gridCoordinates(1);
215  gridCoordinates(0) = gridCoordinates(0) * cellSizeWidth;
216  gridCoordinates(1) = gridCoordinates(1) * cellSizeHeight;
217 
218  // Normalize to 1.0.
219  gridCoordinates = centres.col(i) - gridCoordinates;
220  gridCoordinates(0) /= cellSizeWidth;
221  gridCoordinates(1) /= cellSizeHeight;
222 
223  if (normalize)
224  centreCoordinates = gridCoordinates;
225 
226  if (version == 1)
227  {
228  // Fill elements in the grid.
229  for (size_t k = 0; k < numBoxes; k++)
230  {
231  size_t s = 5 * k;
232  outputTemp(arma::span(gridX), arma::span(gridY),
233  arma::span(s, s + 1)) = centreCoordinates;
234  outputTemp(arma::span(gridX), arma::span(gridY),
235  arma::span(s + 2, s + 3)) = widthAndHeight.col(i);
236  outputTemp(gridX, gridY, s + 4) = 1.0;
237  }
238  outputTemp(gridX, gridY, 5 * numBoxes + labels.col(i)(0)) = 1;
239  }
240  else
241  {
242  size_t s = 0;
243  if (boundingBoxOffset.count({gridX, gridY}))
244  {
245  s = boundingBoxOffset[{gridX, gridY}] + 1;
246  boundingBoxOffset[{gridX, gridY}]++;
247  }
248  else
249  boundingBoxOffset.insert({{gridX, gridY}, s});
250 
251  if (s > numBoxes)
252  continue;
253 
254  size_t bBoxOffset = (5 + numClasses) * s;
255  outputTemp(arma::span(gridX), arma::span(gridY),
256  arma::span(bBoxOffset, bBoxOffset + 1)) = centreCoordinates;
257  outputTemp(arma::span(gridX), arma::span(gridY),
258  arma::span(bBoxOffset + 2,
259  bBoxOffset + 3)) = widthAndHeight.col(i);
260  outputTemp(gridX, gridY, bBoxOffset + 4) = 1.0;
261  outputTemp(gridX, gridY, bBoxOffset + 5 + labels.col(i)(0)) = 1;
262  }
263  }
264  }
