384  void TrainTestSplit(DatasetX& dataset,
385  std::deque<arma::vec>& labels,
386  DatasetX& /* trainFeatures */,
387  arma::field<arma::vec>& /* trainLabels */,
388  DatasetX& /* validFeatures */,
389  arma::field<arma::vec>& /* validLabels */,
390  const double validRatio,
391  const bool shuffle)
392  {
393  const size_t validSize = static_cast<size_t>(dataset.n_cols * validRatio);
394  const size_t trainSize = dataset.n_cols - validSize;
395 
396  arma::uvec order = arma::linspace<arma::uvec>(0, dataset.n_cols - 1,
397  dataset.n_cols);
398  if (shuffle)
399  order = arma::shuffle(order);
400 
401  if (trainSize > 0)
402  {
403  trainLabels.set_size(1, trainSize);
404  trainFeatures = dataset.cols(order.subvec(0, trainSize - 1));
405 
406  // Field type has fixed size so we can't use span and assignment
407  // operator.
408  for (size_t i = 0; i < trainSize; i++)
409  trainLabels(0, i) = labels[i];
410  }
411 
412  if (validSize <= dataset.n_cols)
413  {
414  validFeatures = dataset.cols(order.subvec(trainSize,
415  dataset.n_cols - 1));
416  validLabels.set_size(1, validSize);
417  for (size_t i = trainSize; i < dataset.n_cols; i++)
418  validLabels(0, i - trainSize) = labels[i];
419  }
420  return;
421  }

434  void TrainTestSplit(DatasetX& dataset,
435  std::deque<arma::vec>& labels,
436  DatasetX& /* trainFeatures */,
437  arma::mat& /* trainLabels */,
438  DatasetX& /* validFeatures */,
439  arma::mat& /* validLabels */,
440  const double validRatio,
441  const bool shuffle)
442  {
443  // Calculate number of objects in the image.
444  size_t numberOfObjects = labels[0].n_rows;
445  DatasetY labelsTemp(numberOfObjects, labels.size());
446 
447  for (size_t i = 0; i < labels.size(); i++)
448  labelsTemp.col(i) = labels[i];
449 
450  DatasetX completeDataset = arma::join_cols(dataset, labelsTemp);
451  mlpack::data::Split(completeDataset, trainFeatures, validFeatures,
452  validRatio, shuffle);
453 
454  // Features are all rows except the last 5 rows which correspond
455  // to bounding box.
456  trainLabels = trainFeatures.rows(trainFeatures.n_rows -
457  numberOfObjects, trainFeatures.n_rows - 1);
458  trainFeatures = trainFeatures.rows(0, trainFeatures.n_rows -
459  numberOfObjects - 1);
460 
461  validLabels = validFeatures.rows(validFeatures.n_rows -
462  numberOfObjects, validFeatures.n_rows - 1);
463  validFeatures = validFeatures.rows(0, validFeatures.n_rows -
464  numberOfObjects - 1);
465  return;
466  }
